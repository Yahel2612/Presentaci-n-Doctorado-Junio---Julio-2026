\section{Motivación}

\begin{frame}{¿Por qué generalizar la variación acotada?}
\framesubtitle{Contexto y pregunta central}

\begin{block}{Punto de partida: caso clásico}
El Teorema de Jordan garantiza que toda $f:[a,b]\to\mathbb{R}$ de
variación acotada se descompone como diferencia de dos funciones crecientes.
\end{block}

\medskip

\begin{block}{La pregunta}
¿Qué ocurre si $f:[a,b]\to X$, donde $(X,\|\cdot\|,\Xp)$ es un
\textbf{espacio normado ordenado}?
\end{block}

\medskip

\begin{alertblock}{Respuesta (adelanto)}
Las propiedades clásicas dependen de la \textbf{geometría del cono} $\Xp$:
normalidad, solidez, ser generador o extendible.
\end{alertblock}

\end{frame}

\section{Variación acotada para funciones vectoriales}

%──────────────────────────────────────────────────────────────────────────────

\begin{frame}{Funciones vectoriales sobre ENOs}
\framesubtitle{El nuevo objeto de estudio}

\begin{definition}
Sea $(X,\|\cdot\|,\Xp)$ un ENO. Una función
$f:[a,b]\to X$ se llama \textbf{función vectorial}.
\end{definition}

\medskip

\begin{tikzpicture}[scale=0.9]
    \draw[thick,blue] (0,0) -- (3,0);
    \draw[fill=blue] (0,0) circle (0.07) node[below]{$a$};
    \draw[fill=blue] (3,0) circle (0.07) node[below]{$b$};
    \node[blue,below] at (1.5,-0.45){\small$[a,b]$};
    \draw[fill=gray!20,rounded corners=4pt,draw=gray!60] (5,-0.8) rectangle (9,1.8);
    \node at (7,0.5){$(X,\|\cdot\|,\Xp)$};
    \draw[->,green!60!black,thick] (3.2,0.1)
          to[out=30,in=160]
          node[midway,above]{\textcolor{green!60!black}{$f$}} (4.9,-0.2);
\end{tikzpicture}

\medskip
La pregunta: ¿cómo generalizar la variación total cuando el codominio
tiene \emph{estructura de orden}?

\end{frame}

%──────────────────────────────────────────────────────────────────────────────

\begin{frame}{Tres nociones de variación acotada}
\framesubtitle{Las generalizaciones no son equivalentes en general}

Al generalizar $f:[a,b]\to X$, surgen \textbf{tres nociones distintas}:

\medskip
\begin{enumerate}
    \item \textbf{Variación fuerte:}
          suma de normas de incrementos sobre toda partición.

    \item \textbf{Variación débil:}
          variación de $\phi\circ f$ en $\mathbb{R}$ para todo $\phi\in X^*$.

    \item \textbf{Variación en orden}
          (solo en espacios de Riesz):
          suma acotada superiormente en el orden de $X$.
\end{enumerate}

\medskip
\alert{Pregunta:} ¿cuáles implicaciones se cumplen entre ellas?

\smallskip
{\small Las siguientes diapositivas detallan cada noción.}

\end{frame}

%──────────────────────────────────────────────────────────────────────────────

\begin{frame}{Variación acotada fuerte: diagrama}
\framesubtitle{Generalización directa de la norma}

\begin{itemize}
    \item \textbf{V.a.\ fuerte:}
    \begin{block}{}
        $\Var_a^b(f,X)
        := \sup\Bigl\{\sum_{i=1}^n\|f(t_i)-f(t_{i-1})\| :
        \mathcal{P}\in\wp[a,b]\Bigr\} < \infty$
    \end{block}
    \begin{tikzpicture}[scale=0.85]
        \draw[thick,blue] (0,0) -- (3,0);
        \draw[fill=gray!20,rounded corners=4pt,draw=gray!60]
              (4.5,-0.8) rectangle (10,1.8);
        \node[gray!60] at (7.25,1.5){\small$X$};
        \draw[->,green!60!black] (1.5,0.2) to[bend left=25] (6,0.7);
        \draw[->,green!60!black] (1,0.2) to[bend left=25] (7,1.3);
        \draw[fill] (0,0) circle (0.07) node[below]{$a$};
        \draw[fill] (1,0) circle (0.07) node[below]{$t_{i-1}$};
        \draw[fill] (1.5,0) circle (0.07) node[below]{$t_i$};
        \draw[fill] (3,0) circle (0.07) node[below]{$b$};
        \draw[fill] (6,0.7) circle (0.06) node[below]{\tiny$f(t_{i-1})$};
        \draw[fill] (7,1.3) circle (0.06) node[below]{\tiny$f(t_i)$};
        \draw[dashed,red,thick] (6,0.7) -- (7,1.3);
        \node[right,red] at (6.5,1.05){\tiny$\|f(t_i)-f(t_{i-1})\|$};
    \end{tikzpicture}
\end{itemize}

\end{frame}

%──────────────────────────────────────────────────────────────────────────────

\begin{frame}{Variación acotada débil}
\framesubtitle{Definición vía funcionales continuos}

\begin{definition}
$f:[a,b]\to X$ es de \textbf{variación acotada débil} si para todo
funcional continuo $\phi\in X^*$ la composición
$\phi\circ f:[a,b]\to\mathbb{R}$ es de variación acotada.
\end{definition}

    \foreach \x/\l in {-3.16/t_0,-1.8/t_1,-0.5/t_2,1.39/t_3,3.3/t_4}

\end{tikzpicture}

\column{.36\textwidth}
\small
$\phi\in X^*$ proyecta $f$ sobre $\mathbb{R}$.

\medskip
Se mide la variación de $\phi\circ f$ con la norma usual de $\mathbb{R}$.

\medskip
\textcolor{purple}{La variación débil} controla las
oscilaciones de cada proyección.

\end{columns}

\end{frame}

%──────────────────────────────────────────────────────────────────────────────

\begin{frame}{Variación acotada en orden}
\framesubtitle{Solo tiene sentido en espacios de Riesz}

\begin{definition}
Sea $(X,\|\cdot\|)$ un espacio de Riesz. Una función $f:[a,b]\to X$
es de \textbf{variación acotada en orden} si existe $M\in X$ tal que
\[
V_f^o := \sup\Bigl\{\sum_{i=1}^n|f(t_i)-f(t_{i-1})| :
\mathcal{P}\in\wp[a,b]\Bigr\} \le M.
\]
\end{definition}

\medskip

\begin{theorem}
Sea $X$ un retículo de Banach. Entonces:
\begin{enumerate}
    \item Toda función creciente es de v.a.\ en orden.
    \item V.a.\ en orden $\Rightarrow$ v.a.\ débil.
\end{enumerate}
\end{theorem}

\medskip
\alert{Nota:} la variación en orden solo aplica cuando $|x|$ tiene
sentido, es decir, en espacios de Riesz.

\end{frame}

\section{Funciones crecientes}

%──────────────────────────────────────────────────────────────────────────────

\begin{frame}{Funciones crecientes en un ENO}
\framesubtitle{Generalización del monotonismo}

\begin{definition}
Sean $(X,\|\cdot\|,\Xp)$ un ENO y $f:[a,b]\to X$ vectorial.
$f$ es \textbf{creciente} si $x < y$ implica $f(x)\le f(y)$.
\end{definition}

\medskip

\begin{alertblock}{Pregunta clave}
¿Toda función creciente es de variación acotada?

\smallskip
En $\mathbb{R}$: \textbf{sí}. En un ENO general: \textbf{depende del cono}.
\end{alertblock}

% IDEA ORAL: Esta es la pregunta central de la sección. Los resultados que
% siguen responden exactamente cuándo sí y cuándo no.

\end{frame}

%──────────────────────────────────────────────────────────────────────────────

\begin{frame}{Ejemplo: creciente con v.a.\ débil pero no fuerte}
\framesubtitle{Las hipótesis no son decorativas}

\begin{block}{Ejemplo}
Sea $F:[0,1]\to L_\infty[0,1]$ dada por $F(t)=\chi_{[0,t]}$
con $\Xp=\{f\in L_\infty[0,1]:f\ge 0 \text{ c.t.p.}\}$.

\medskip
$F$ es \textbf{creciente} y de \textbf{v.a.\ débil}, pero
\textbf{no} es de v.a.\ fuerte.
\end{block}

\begin{tikzpicture}[scale=0.85]
    \draw[-] (-1,0)--(3,0) node[right]{$x$};
    \draw[-] (0,-0.5)--(0,2) node[above]{$y$};
    \draw[-,blue,thick] (0,1)--(1,1);
    \draw[dashed,blue] (1,1)--(1,0);
    \node at (1,-0.9){\small$\chi_{[0,t_1]}$};
    \node[blue] at (1,-0.4){\small$t_1$};
    \draw[-] (5,0)--(9,0) node[right]{$x$};
    \draw[-] (6,-0.5)--(6,2) node[above]{$y$};
    \draw[-,red,thick] (6,1)--(8,1);
    \draw[dashed,red] (8,1)--(8,0);
    \node at (8,-0.9){\small$\chi_{[0,t_2]}$};
    \node[red] at (8,-0.4){\small$t_2$};
\end{tikzpicture}

\end{frame}

%──────────────────────────────────────────────────────────────────────────────

\begin{frame}{Contraejemplo: creciente sin ninguna variación acotada}
\framesubtitle{Por qué esta hipótesis importa}

\begin{block}{Contraejemplo}
Sea $X=C_{\mathbb{R}}^1[0,1]$ con $\|f\|=\|f\|_\infty+\|f'\|_\infty$
y $\Xp=\{f:f(0)=0,\,f(t)\ge 0\}$.

\medskip
Existe una función $f:[0,1]\to X$ \textbf{creciente} que
\textbf{no} es de v.a.\ débil (y por tanto tampoco de v.a.\ fuerte).
\end{block}

\textbf{Construcción.}
Se define
$f(t) = x_n$ para $t\in[1-\tfrac{1}{n},1-\tfrac{1}{n+1})$
con $x_n(s)=-s^n$, y $f(1)=\bar{0}$.

El funcional $D:X\to\mathbb{R}$, $D(f)=f'(1)$, satisface
$|D(f)|\le\|f\|$, pero
$\Var_a^b(D\circ f,P_m)\ge m+1$ para $P_m=\{0,1-\tfrac{1}{m+1},1\}$.

Por tanto, $D\circ f\notin BV[0,1]$.

\end{frame}

%──────────────────────────────────────────────────────────────────────────────

\begin{frame}{Equivalencia: normalidad y variación débil}
\framesubtitle{El cono normal es la condición correcta}

\begin{block}{Lema (equivalencia)}
Sea $X$ un ENO. Son equivalentes:
\begin{enumerate}
    \item $\Xp$ es \textbf{normal}.
    \item Para toda $f:[a,b]\to X$ \textbf{creciente}, $f$ es de v.a.\ \textbf{débil}.
    \item Toda sucesión creciente $\{x_n\}$ en $X$ converge
          si y solo si tiene una subsucesión convergente.
\end{enumerate}
\end{block}

\medskip

\begin{alertblock}{Lectura del resultado}
La normalidad del cono es \emph{exactamente} la condición que hace que
el monotonismo implique variación débil.
El contraejemplo anterior muestra que sin normalidad esta implicación falla.
\end{alertblock}

% IDEA ORAL: Insistir en que el lema dice "si y solo si". No es solo
% una condición suficiente: es la condición caracterizadora exacta.

\end{frame}

%──────────────────────────────────────────────────────────────────────────────

\begin{frame}{Cono extendible y variación acotada fuerte}
\framesubtitle{Resultado positivo bajo una hipótesis más fuerte}

\begin{theorem}
Si $X$ es un ENO con cono \textbf{extendible}, entonces toda función
creciente $f:[a,b]\to X$ es de \textbf{variación acotada fuerte}.
\end{theorem}

\medskip

\begin{block}{Diagrama de implicaciones}
\centering
\medskip
$\Xp$ extendible
$\;\Rightarrow\;$
creciente $\Rightarrow$ v.a.\ fuerte

\medskip
$\Xp$ normal
$\;\Rightarrow\;$
creciente $\Leftrightarrow$ v.a.\ débil

\medskip
(Sin hipótesis sobre el cono: ninguna implicación garantizada)
\end{block}

\end{frame}

\section{Trabajo futuro}

\begin{frame}{Integración de Riemann-Stieltjes vectorial}
\framesubtitle{Siguiente línea de investigación}

\begin{definition}
Una función $f:[a,b]\to X$ acotada es \textbf{integrable en el sentido
Riemann-Stieltjes} respecto a $g:[a,b]\to\mathbb{R}$ si existe $A\in X$
tal que, dado $\epsilon>0$, existe $\delta>0$ de modo que para toda
partición $\{(I_i,\xi_i):i=1,\ldots,n\}$ de $[a,b]$ con norma menor
que $\delta$, se cumple
\[
\Bigl\|\sum_{i=1}^n f(\xi_i)\bigl(g(t_i)-g(t_{i-1})\bigr) - A\Bigr\|
< \epsilon.
\]
\end{definition}

\medskip

\begin{alertblock}{Conexión con los resultados obtenidos}
Si $g$ es de v.a.\ y $f$ es continua, ¿basta con que $g$ sea de v.a.\
débil para garantizar la existencia de la integral?

El papel del cono $\Xp$ en la integrabilidad es la siguiente pregunta abierta.
\end{alertblock}

\end{frame}

\begin{frame}{Propiedades del espacio $l^1$}
\framesubtitle{Material de respaldo}
\begin{enumerate}
    \item Es un espacio de Schur
          (convergencia débil implica convergencia en norma).
    \item Todo intervalo ordenado cerrado $[u,v]$ en $l^1$ es
          \textbf{compacto}.
\end{enumerate}

\medskip

Estas propiedades son relevantes para aplicar teoremas de punto fijo
en $l^1$ con el orden coordinada a coordenada.
\end{frame}

%──────────────────────────────────────────────────────────────────────────────

\begin{frame}{Teoremas de punto fijo en ENOs}
\framesubtitle{Aplicaciones directas — material de respaldo}

\begin{block}{Teorema 1 (espacio de Schur)}
Sea $X$ un ENO con la propiedad de Schur, $u_0,v_0\in X$ con
$u_0<v_0$.
Si $A:[u_0,v_0]\to[u_0,v_0]$ es subcontinuo creciente y existe $B$
débilmente compacto con $A([u_0,v_0])\subseteq B$,
entonces $A$ tiene un punto fijo minimal y maximal.
\end{block}

\begin{block}{Teorema 2 (intervalos compactos)}
Sea $X$ un ENO donde todo intervalo ordenado $[a,b]$ es compacto,
$u_0,v_0\in X$ con $u_0<v_0$.
Si $A:[u_0,v_0]\to[u_0,v_0]$ es creciente,
entonces $A$ tiene un punto fijo minimal y maximal.
\end{block}

\end{frame}

\begin{frame}{El espacio $C^1([a,b],X)$ como ENO}
\framesubtitle{Material de respaldo}

Sea $X$ un ENO con $\Xp$ normal y $E^1=C^1([a,b],X)$.

Se define $E_+^1=\{f\in E^1:f(t)\ge 0,\;f'(t)\ge 0\;\forall t\}$
y $\|f\|_\infty=\sup\{\|f(t)\|_X:t\in[a,b]\}$.

\medskip

\textbf{Propiedades:}
\begin{enumerate}
    \item Si $\Xp$ es normal, entonces $E_+^1$ es normal.
    \item Si en $X$ todos los intervalos ordenados cerrados son compactos,
          entonces en $E^1$ todos los intervalos ordenados cerrados son
          compactos.

\begin{frame}{Demostración: Teorema en $l^1$}
\framesubtitle{Material de respaldo}

\begin{theorem}
Sea $f:[a,b]\to l^1$ con coordenadas $f_j$. Si $f\in\BVX$, entonces
$f_j\in BV[a,b]$ para todo $j\in\mathbb{N}$ y
$V(f,[a,b])=\sum_{j=1}^\infty V(f_j,[a,b])$.
\end{theorem}

\textit{Idea de la demostración.}
La inclusión $\sum V(f_j)\le V(f)$ se obtiene de la desigualdad
$\sum_i|f_j(t_i)-f_j(t_{i-1})|\le\sum_i\|f(t_i)-f(t_{i-1})\|_1$.

Para la otra desigualdad, si $V(f)<\sum V(f_j)$, existe $n_0$ tal que
$V(f)<\sum_{j=1}^{n_0}V(f_j)$; tomando $\epsilon$ adecuado y una
partición común $P=\bigcup_j P_j$ se obtiene una contradicción por
intercambio de sumas finitas.
\end{frame}

\begin{frame}{Demostración: normalidad del cono (lema)}
\framesubtitle{Material de respaldo}

\textbf{Lema.} Sea $X$ un ENO. Son equivalentes:
(i) $\Xp$ normal; (ii) toda sucesión creciente tiene subsucesión
convergente implica convergencia de la sucesión.

\medskip
\textit{($\Rightarrow$)} Sea $\{x_n\}$ creciente con subsucesión $\{x_{n_k}\}\to x^*$.
Para $n\ge n_k$: $0\le x^*-x_n\le x^*-x_{n_k}$, y por normalidad
$\|x^*-x_n\|\le M\|x^*-x_{n_k}\|\to 0$.

\textit{($\Leftarrow$)} Si $\Xp$ no es normal, existen $0\le x_n\le y_n$
con $\|x_n\|>2^n\|y_n\|$. Sea $z_n=x_n/\|x_n\|$, $v_n=y_n/(2^n\|y_n\|)$.
Se tiene $0\le z_n\le v_n$ y $\sum\|v_n\|=1$, de donde $\sum v_n\to v$.
La sucesión $w_n$ construida en términos de sumas parciales tiene
una subsucesión convergente pero no converge. Contradicción.
\end{frame}
