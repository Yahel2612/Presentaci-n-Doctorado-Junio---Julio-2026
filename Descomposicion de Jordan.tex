\begin{frame}{}
\begin{block}{}
\centering\large
Descomposición de Jordan para funciones vectoriales
\end{block}
\end{frame}

%──────────────────────────────────────────────────────────────────────────────

\begin{frame}{Definición: descomposición de Jordan}
\framesubtitle{Generalización del teorema clásico}

\begin{definition}
Un ENO $X$ satisface la \textbf{descomposición de Jordan} si
para toda $f:[a,b]\to X$ de variación acotada existen
$g,h:[a,b]\to X$ funciones crecientes tales que $f=g-h$.
\end{definition}

\medskip

\begin{block}{Lema}
Si $f:[a,b]\to X$ cumple la descomposición de Jordan,
entonces $f$ es \emph{acotada en orden}.
\end{block}

% IDEA ORAL: Este lema dice que la descomposición de Jordan tiene un
% costo: necesita que la función sea "controlable" por el orden.

\end{frame}

%──────────────────────────────────────────────────────────────────────────────

\begin{frame}{Condición necesaria: el cono debe ser generador}
\framesubtitle{Sin esta hipótesis, la descomposición puede fallar}

\begin{theorem}
Sea $X$ un ENO. Si $\Xp$ \emph{no} es generador, entonces existe
$f:[a,b]\to X$ de variación acotada que \textbf{no admite}
descomposición de Jordan.
\end{theorem}

\medskip
\textbf{Prueba.} Sea $x_0\in X\setminus(\Xp-\Xp)$. Se define
\[
f(t)=\begin{cases}
  2x_0, & t=b,\\
  x_0,  & t=a,\\
  0,    & t\in(a,b).
\end{cases}
\]
Si existieran $g,h$ crecientes con $f=g-h$, entonces
$x_0 = (g(b)-g(a))-(h(b)-h(a)) \in \Xp-\Xp$. Contradicción. \qed

\end{frame}

%──────────────────────────────────────────────────────────────────────────────

\begin{frame}{Ejemplo: $L_1[0,1]$ con cono generador}
\framesubtitle{Un espacio que sí admite descomposición de Jordan}

\begin{block}{}
$(L_1[0,1],\|\cdot\|_1,\Xp)$ con $\Xp=\{f\in L_1[0,1]:f\ge 0$ c.t.p.$\}$
tiene cono generador, pues $f = f^{+} - f^{-}$ para toda $f\in L_1[0,1]$.

\smallskip
En particular, admite la descomposición de Jordan.
\cite{Gou2004}
\end{block}

\end{frame}

%──────────────────────────────────────────────────────────────────────────────

\begin{frame}{Condición suficiente: cono sólido}
\framesubtitle{La solidez garantiza la descomposición}

\begin{definition}
Se dice que $\mu\in\Xp$ es una \textbf{cota superior en orden de la esfera}
$S(X)=\{x\in X:\|x\|=1\}$ si $x\le\|x\|\,\mu$ para todo $x\in X$.
\end{definition}

\medskip

\begin{theorem}
Si $X$ tiene una cota superior en orden $\mu$ de $S(X)$, entonces
toda $f:[a,b]\to X$ de variación acotada admite descomposición de Jordan.
\end{theorem}

\end{frame}

%──────────────────────────────────────────────────────────────────────────────

\begin{frame}{Solidez y descomposición de Jordan}
\framesubtitle{Equivalencia clave y consecuencia}

\begin{block}{Lema (clave)}
$S(X)$ es acotada superiormente en orden $\iff$ $\Xp$ es \textbf{sólido}.
\end{block}

\medskip

\begin{alertblock}{Consecuencia}
$\Xp$ sólido $\Longrightarrow$ descomposición de Jordan para toda función de v.a.
\end{alertblock}

\end{frame}

%──────────────────────────────────────────────────────────────────────────────

\begin{frame}{Descomposición explícita cuando $\Xp$ es sólido}
\framesubtitle{Construcción de las funciones crecientes}

Sea $\mu\in\operatorname{int}(\Xp)$ cota superior de $S(X)$ y
$f:[a,b]\to X$ de variación acotada. Se definen

\[
M_f(t) = \tfrac{1}{2}\Bigl[\Var_a^t(f,X)\,\mu + f(t) - f(a)\Bigr],
\]
\[
N_f(t) = \tfrac{1}{2}\Bigl[\Var_a^t(f,X)\,\mu - f(t) + f(a)\Bigr].
\]

\medskip

\begin{theorem}[\cite{perez2022jordan}]
$M_f$ y $N_f$ son funciones \textbf{crecientes} de variación acotada y
$f(t) = M_f(t) - N_f(t) + f(a)$.
\end{theorem}

% IDEA ORAL: Esta es la construcción explícita análoga al caso real.
% Las funciones M_f y N_f generalizan la parte positiva y negativa.

\end{frame}

%──────────────────────────────────────────────────────────────────────────────

\begin{frame}{Estimaciones de la descomposición}
\framesubtitle{Control de la variación de $M_f$ y $N_f$}

\begin{theorem}[\cite{perez2022jordan}]
Con las definiciones anteriores:
\begin{enumerate}
    \item $\Var_a^b(M_f,X),\;\Var_a^b(N_f,X) \le
          \dfrac{1+\|\mu\|}{2}\,\Var_a^b(f,X)$.
    \item $\Var_a^b(M_f+N_f,X) = \Var_a^b(f,X)\,\|\mu\|$.
    \item $\Var_a^b(M_f-N_f,X) = \Var_a^b(f,X)$.
\end{enumerate}
\end{theorem}

\medskip
\begin{block}{Lectura}
Las funciones $M_f$ y $N_f$ tienen variación controlada por la de $f$.
Su diferencia $M_f - N_f$ conserva \emph{exactamente} la variación total.
\end{block}

\end{frame}

%──────────────────────────────────────────────────────────────────────────────

\begin{frame}{Consecuencias de la descomposición}
\framesubtitle{Norma en $\BVX$ y estructura algebraica}

\begin{theorem}[\cite{perez2022jordan}]
\begin{enumerate}
    \item $\|f\|_{BV} = \|M_f-N_f\|_{BV} + \|f(a)\|$.
    \item Si $X$ es un álgebra, entonces $\BVX$ es un álgebra.
\end{enumerate}
\end{theorem}

\medskip

\begin{alertblock}{Significado}
La norma BV se descompone canónicamente a través de la descomposición
de Jordan. Cuando $X$ es álgebra, esta estructura se transfiere a $\BVX$.
\end{alertblock}

\begin{frame}{Ejemplo: $l^1$ sin cono sólido, con descomposición de Jordan}
\framesubtitle{La condición sólido no es necesaria}

\begin{block}{}
$(l^1,\|\cdot\|_1,l^1_+)$ con
$l^1_+ = \bigl\{\{a_n\}\in l^1 : a_n\ge 0 \text{ para todo } n\in\mathbb{N}\bigr\}$
tiene cono generador pero \emph{no sólido}.

\smallskip
Aun así, \textbf{admite la descomposición de Jordan}.
\end{block}

{\small\textbf{Argumento.}
Sea $f:[a,b]\to l^1$ de variación acotada con coordenadas $f_j$.
Por el teorema del caso real, $f_j = V(f_j,[a,\cdot]) - [V(f_j,[a,\cdot])-f_j]$.
Dado que $\sum_j V(f_j,[a,b])\le V(f,[a,b])<\infty$,
las sumas $f_2=\{V(f_j,[a,\cdot])\}$ y $f_1=\{V(f_j,[a,\cdot])-f_j\}$
pertenecen a $l^1$ y $f = f_2 - f_1$.}

\medskip
\begin{alertblock}{Lección}
La solidez es \emph{suficiente}, pero \emph{no necesaria}.
El cono generador es la condición \emph{necesaria} fundamental.
\end{alertblock}

\end{frame}
