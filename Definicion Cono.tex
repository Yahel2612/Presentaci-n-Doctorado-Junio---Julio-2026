\section{Espacios normados ordenados}

\begin{frame}{¿Qué es un espacio normado ordenado?}
\framesubtitle{La estructura que permite hablar de orden y norma juntos}

\begin{columns}
    \column{.52\textwidth}
    \begin{definition}[\cite{cone2007}]
    Sea $(X,\|\cdot\|)$ un espacio normado y $\Xp\subset X$
    cerrado, no vacío y \textbf{propio}.
    $\Xp$ es un \textbf{cono} si para todo $x,y\in\Xp$ y $\lambda\ge 0$:
    \begin{itemize}
        \item $x+y\in\Xp$,
        \item $\lambda x\in\Xp$,
        \item $-x,x\in\Xp\Rightarrow x=\mathbf{0}$.
    \end{itemize}
    \end{definition}

    A $(X,\|\cdot\|,\Xp)$ se le llama
    \textbf{espacio normado ordenado} (ENO).

    \column{.42\textwidth}
    \begin{tikzpicture}[scale=0.68]
        \draw[color=blue,thick] (0,-1)--(-1,1)--(1,1)--(0,-1);
        \draw[color=blue,thick] (-1,1)--(0,1.5)--(1,1);
        \draw[dashed,color=blue] (0,-1)--(0,1.5);
        \draw[color=red,thick] (0,-1)--(-1,-3)--(1,-3)--(0,-1);
        \draw[dashed,color=red] (-1,-3)--(0,-2.5)--(1,-3);
        \draw[dashed,color=red] (0,-1)--(0,-2.5);
    \end{tikzpicture}

    \smallskip
    \textcolor{blue}{$\Xp$}: cono positivo. \quad
    \textcolor{red}{$-\Xp$}: cono negativo. \\[2pt]
    $\Xp\cap(-\Xp)=\{\mathbf{0}\}$.
\end{columns}

\end{frame}

\begin{frame}{La relación de orden inducida por el cono}
\framesubtitle{El cono define cómo comparar elementos}

\begin{theorem}
En un ENO $(X,\|\cdot\|,\Xp)$, la relación
$a\le b \;\overset{\text{def}}{\iff}\; b-a\in\Xp$
es un \textbf{orden parcial compatible} con la estructura lineal:
\begin{enumerate}[(1)]
    \item $a\le a$ \hfill (reflexividad)
    \item $a\le b$ y $b\le a \Rightarrow a=b$ \hfill (antisimetría)
    \item $a\le b$ y $b\le c \Rightarrow a\le c$ \hfill (transitividad)
    \item $a\le b \Rightarrow a+c\le b+c$ \hfill (traslación)
    \item $a\le b \Rightarrow \lambda a\le\lambda b$ para $\lambda\ge 0$ \hfill (escala)
\end{enumerate}
\end{theorem}

\smallskip
Convenio: $a<b$ si $a\le b$ y $a\ne b$.

\end{frame}
