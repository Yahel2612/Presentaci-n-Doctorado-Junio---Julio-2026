\section{Variación acotada clásica}



%------------------------------------------
\begin{frame}{Particiones y suma de la variación}
Comencemos recordando qué son las particiones y la suma de la variación con respecto de una partición.

\medskip

Una \textbf{partición} de \([a,b]\) es un conjunto ordenado de puntos
\[
\mathcal{P}: \{ a = x_0 < x_1 < \cdots < x_n = b \} .
\]

\medskip

Dada una función \(f:[a,b]\to \mathbb{R}\), la \textbf{variación de \(f\) respecto a la partición \(\mathcal{P}\)} es
\[
V(f,\mathcal{P})=\sum_{k=1}^n \bigl|f(x_k)-f(x_{k-1})\bigr|.
\]
\end{frame}

%------------------------------------------
\begin{frame}{Variación total y variación acotada}
Con esto ya podemos dar la definición de variación acotada en el caso real.

\begin{block}{}
    La variación total de \(f\) en \([a,b]\) se define como
\[
V_a^b(f)=\sup\{ V(f,\mathcal{P}) : \mathcal{P} \text{ partición de } [a,b]\}.
\]

Si \(V_a^b(f)<\infty\),
decimos que \(f\) es de \textbf{variación acotada} en \([a,b]\).
\end{block}

\medskip

Denotaremos por \(BV[a,b]\) al conjunto de funciones de variación acotada en \([a,b]\).
\end{frame}

\begin{frame}{Propiedades y ejemplos clásicos}
\framesubtitle{Ejemplos básicos de funciones en \(BV[a,b]\)}



\begin{enumerate}
    \item \textbf{Función constante}\\[2mm]
    Si \(f(x)\equiv c\), entonces \(V(f,\mathcal{P})=0\) para toda partición, por tanto
    \[
    V_a^b(f)=0.
    \]

    \item \textbf{Función monótona}\\[2mm]
    Si \(f\) es creciente, independientemente de la partición
    \[
    V(f,\mathcal{P})=f(b)-f(a).
    \]
    Por tanto toda función monótona es de variación acotada.

    \item \textbf{Función Lipschitz}\\[2mm]
    Si \(f\) es Lipschitz con constante \(L\), entonces
    \[
    V_a^b(f)\le L(b-a)<\infty.
    \]
\end{enumerate}
\end{frame}

\begin{frame}{}
    \begin{block}{Lema}
        
    Sea \(f,g \in BV[a,b]\), \(c\in\mathbb{R}\). Entonces:

\begin{enumerate}
    \item \(V_a^b( f) = 0\) si y sólo si $f$ es una constante. 
    \item \(V_a^b(c f) = |c|\,V_a^b(f)\).
    \item \(V_a^b(f+g) \leq V_a^b(f) + V_a^b(g)\).
    \item \(V_a^b(fg) \leq ||f||_{\infty} V_a^b(g) + ||g||_{\infty} V_a^b(f)\).
    \item \(V_a^b(| f|) = \,V_a^b(f)\).
    \item Si  \(c\in(a,b)\), entonces $V_a^b(f)=V_a^c(f)+V_c^b(f).$
\end{enumerate}
    \end{block}
\end{frame}
\begin{frame}{}
    \begin{block}{Teorema}
         \( BV[a,b]\) es un espacio completo (espacio de Banach) con las operaciones usuales  de funciones y la norma $ || f ||_{BV}= |f(a)| + V_a^b f $.
    \end{block}
    \begin{block}{\textbf{Teorema de Descomposición de Jordan}}
         
    Una función \(f:[a,b]\to\mathbb{R}\) es de variación acotada si y sólo si
puede escribirse como
\[
f = g-h,
\]
donde \(g\) y \(h\) son funciones crecientes en \([a,b]\).
    \end{block}
\end{frame}
