\begin{frame}
\begin{itemize}
	 \item \textbf{Ejemplo 1}
	\begin{block}{}%\arrowvert
	Sean $X=C[a,b]$ y su cono  $P_{1} =\left\lbrace f \in C[a,b] \mid f(x) \geq 0 \right\rbrace  $.\\  Todo $ f \in X   $ tal que $ f(x)>0 $ para  $ x \in [a,b] $ es unidad de orden\\ y $ X $ es Arquimediano.
\end{block}
	\item 
	 \textbf{Ejemplo 2}
		\begin{block}{}
		Sea $ X = \left\lbrace \left\lbrace x_n \right\rbrace  : x_n \in \mathbb{R} \right\rbrace  $ y $ P=  \left\lbrace \left\lbrace x_n \right\rbrace \in X : x_n \geq 0 \right\rbrace  $.\\
		$ X $ no tiene unidades de $ P- $orden.\\
	\end{block}
  
\end{itemize}

\end{frame}
